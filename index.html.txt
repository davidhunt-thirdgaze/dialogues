  <title>Discussions Of Moira And Lee</title>
  <style>
    body { font-family: Arial, sans-serif; padding: 2rem; }
    h1 { font-size: 2em; }
    .line { margin-top: 10px; padding: 8px; background: #f9f9f9; border-left: 3px solid #888; }
  </style>
</head>
<body>
  <h1>Discussions Of Moira And Lee</h1>
  <div id="chatContainer"></div>

  <script>
    async function fetchNextMessage() {
      const res = await fetch('/discuss');
      const data = await res.json();
      if (data.message) {
        addMessageToChat(data.message);
      }
    }

    function addMessageToChat(msg) {
      const container = document.getElementById('chatContainer');
      const div = document.createElement('div');
      div.className = 'line';
      div.textContent = msg;
      container.appendChild(div);

      // Remove oldest if over limit
      const maxLines = 20;
      while (container.children.length > maxLines) {
        container.removeChild(container.firstChild);
      }
    }

    // Initial load
    async function loadHistory() {
      const res = await fetch('/conversation');
      const data = await res.json();
      data.conversation.forEach(msg => addMessageToChat(msg));
    }

    loadHistory();

    // Auto fetch every 30 seconds
    setInterval(fetchNextMessage, 30000);
  </script>
</body>
